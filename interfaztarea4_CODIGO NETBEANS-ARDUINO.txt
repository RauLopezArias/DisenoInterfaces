const char* ssid ="CNT_PEDRO";

int led1= 5;
int pulso1= 0;
int pulso2= 4;
int pulsador1=0;
int pulsador2=0;

void setup() {

pinMode(pulso1,INPUT);
pinMode(pulso2,INPUT);
pinMode(led1,OUTPUT);

  Serial.begin(115200);
  Serial.println("");
  Serial.print("Conectado a ");
  Serial.println(ssid);
  WiFi.mode(WIFI_STA);
  delay(150);
  WiFi.begin(ssid,pass);
  while(WiFi.status() != WL_CONNECTED){
    delay(150);
    Serial.print(".");
  }
  Serial.println("");
  Serial.println(WiFi.localIP());


}

void secuencia1(){
  digitalWrite(led1,HIGH);
delay(500);
digitalWrite(led1,LOW);
delay(500);

delay(500);
digitalWrite(led1,LOW);
}

void secuencia2(){
digitalWrite(led1,HIGH);
delay(500);
digitalWrite(led1,LOW);

}



void loop() {
if(Serial.available()>0){
  mensaje = Serial.read();
  if(mensaje =='B')   secuencia1();
  if(mensaje =='J')   secuencia2();
}
}
